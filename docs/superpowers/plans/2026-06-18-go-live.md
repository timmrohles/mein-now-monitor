```python
"""
Kern-Logik des Sammlers: fragt mein-NOW ab und schreibt einen Snapshot
als JSON-Datei in docs/data/. Der GitHub-Actions-Workflow committed das
zurück ins Repo; das statische Dashboard liest dieselben Dateien.
"""
import json
import time
import urllib.request
import urllib.parse
from datetime import datetime, timezone
from pathlib import Path

try:
    from .queries import ALL_QUERIES, TOP_N    # bei Package-Import
except ImportError:
    from queries import ALL_QUERIES, TOP_N     # bei direktem Aufruf

MEIN_NOW_URL = "https://rest.mein-now.de/now-prod/suche/pc/v1/bildungsangebot"
MEIN_NOW_KEY = "infosysbub-nowsuche"

REPO_ROOT = Path(__file__).resolve().parent
DATA_DIR  = REPO_ROOT / "docs" / "data"


def fetch_query(query, top_n=TOP_N, retries=2):
    """Fragt einen Suchbegriff ab, gibt (total_results, [angebote]) zurück."""
    url = (f"{MEIN_NOW_URL}?sw={urllib.parse.quote(query)}"
           f"&bg=true&size={top_n}")
    req = urllib.request.Request(
        url,
        headers={"X-API-Key": MEIN_NOW_KEY, "User-Agent": "Mozilla/5.0"},
    )
    last_err = None
    for attempt in range(retries + 1):
        try:
            with urllib.request.urlopen(req, timeout=20) as r:
                data = json.loads(r.read())
            total = data.get("page", {}).get("totalElements")
            angebote = data.get("_embedded", {}).get("bildungsangebotDTOList", [])
            return total, angebote
        except Exception as e:                       # noqa: BLE001
            last_err = e
            time.sleep(2 * (attempt + 1))
    raise RuntimeError(f"Abfrage '{query}' fehlgeschlagen: {last_err}")


def _iso_utc_now():
    return datetime.now(timezone.utc).replace(microsecond=0) \
        .isoformat().replace("+00:00", "Z")


def run_snapshot(verbose=False):
    """Kompletter Lauf: alle Begriffe abfragen, Tagesdatei + Index schreiben."""
    DATA_DIR.mkdir(parents=True, exist_ok=True)
    run_at = _iso_utc_now()
    today  = run_at[:10]                     # YYYY-MM-DD

    snapshot = {"run_at": run_at, "queries": {}}
    errors   = []

    for q in ALL_QUERIES:
        try:
            total, angebote = fetch_query(q)
            results = []
            for pos, a in enumerate(angebote, start=1):
                results.append({
                    "position":   pos,
                    "angebot_id": str(a.get("id", "")),
                    "titel":      (a.get("titel") or "")[:300],
                    "anbieter":   (a.get("bildungsanbieter", {}) or {})
                                   .get("name", "")[:200],
                    "termine":    a.get("anzahlTermine"),
                })
            snapshot["queries"][q] = {
                "total_results": total,
                "results":       results,
            }
            if verbose:
                print(f"  ✓ {q}: {total} Treffer, {len(results)} erfasst")
            time.sleep(0.6)
        except Exception as e:               # noqa: BLE001
            errors.append((q, str(e)))
            if verbose:
                print(f"  ✗ {q}: {e}")

    # Tagesdatei schreiben (überschreibt, falls am selben Tag erneut gelaufen)
    day_file = DATA_DIR / f"{today}.json"
    day_file.write_text(
        json.dumps(snapshot, ensure_ascii=False, indent=2),
        encoding="utf-8",
    )

    # Index aktualisieren (idempotent: Datum wird nur einmal aufgenommen)
    index_file = DATA_DIR / "index.json"
    if index_file.exists():
        index = json.loads(index_file.read_text(encoding="utf-8"))
    else:
        index = {"dates": []}
    if today not in index["dates"]:
        index["dates"].append(today)
    index["dates"].sort(reverse=True)
    index_file.write_text(
        json.dumps(index, indent=2),
        encoding="utf-8",
    )

    return {
        "date":         today,
        "run_at":       run_at,
        "queries":      len(snapshot["queries"]),
        "rows_written": sum(len(qd["results"])
                            for qd in snapshot["queries"].values()),
        "errors":       errors,
    }
```

```python
"""
Anomalie-Erkennung: vergleicht die zwei neuesten JSON-Snapshots in
docs/data/ und meldet Auffälligkeiten. Schwellen oben anpassbar.

Nutzung:
    python detect_changes.py
"""
import json
from pathlib import Path

REPO_ROOT = Path(__file__).resolve().parent
DATA_DIR  = REPO_ROOT / "docs" / "data"

# --- Schwellen ---
POSITION_JUMP = 5        # Positionssprung ab X Plätzen = auffällig
TOTAL_PCT     = 0.15     # Trefferzahl-Änderung ab 15 % = auffällig


def load_index():
    idx_file = DATA_DIR / "index.json"
    if not idx_file.exists():
        return []
    return json.loads(idx_file.read_text(encoding="utf-8")).get("dates", [])


def load_snapshot(date):
    return json.loads((DATA_DIR / f"{date}.json").read_text(encoding="utf-8"))


def detect():
    dates = load_index()
    if len(dates) < 2:
        print("Noch nicht genug Snapshots (mindestens 2 nötig).")
        return

    new_date, old_date = dates[0], dates[1]
    new = load_snapshot(new_date)
    old = load_snapshot(old_date)
    print(f"Vergleich: {old['run_at'][:16]}  →  {new['run_at'][:16]}\n")

    alerts = []

    # 1. Trefferzahl-Sprünge
    for q, d in new["queries"].items():
        old_q = old["queries"].get(q)
        if not old_q:
            continue
        old_total = old_q.get("total_results")
        new_total = d.get("total_results")
        if old_total and new_total:
            change = (new_total - old_total) / old_total
            if abs(change) >= TOTAL_PCT:
                alerts.append(
                    f"[TREFFERZAHL] '{q}': {old_total} → {new_total} "
                    f"({change*100:+.0f} %)  — möglicher Index-/Algo-Wechsel"
                )

    # 2. Positionssprünge je Angebot + neue Top-5 + verschwundene
    for q, d in new["queries"].items():
        old_q = old["queries"].get(q, {"results": []})
        old_pos = {r["angebot_id"]: r["position"] for r in old_q.get("results", [])}
        new_pos = {r["angebot_id"]: r["position"] for r in d["results"]}
        new_titel = {r["angebot_id"]: (r["titel"] or "")[:45] for r in d["results"]}
        old_titel = {r["angebot_id"]: (r["titel"] or "")[:45] for r in old_q.get("results", [])}

        for aid, np in new_pos.items():
            op = old_pos.get(aid)
            if op is None:
                if np <= 5:
                    alerts.append(
                        f"[NEU TOP5] '{q}': {new_titel.get(aid,'?')} "
                        f"neu auf Platz {np}"
                    )
            else:
                jump = op - np
                if abs(jump) >= POSITION_JUMP:
                    pfeil = "▲" if jump > 0 else "▼"
                    alerts.append(
                        f"[SPRUNG {pfeil}] '{q}': {new_titel.get(aid,'?')} "
                        f"Platz {op} → {np}"
                    )

        for aid, op in old_pos.items():
            if aid not in new_pos and op <= 10:
                alerts.append(
                    f"[WEG] '{q}': {old_titel.get(aid,'?')} war Platz {op}, "
                    f"heute nicht mehr in Top-{len(new_pos)}"
                )

    if not alerts:
        print("Keine Auffälligkeiten über den Schwellen. Markt stabil.")
    else:
        print(f"{len(alerts)} Auffälligkeit(en):\n")
        for a in alerts:
            print("  " + a)


if __name__ == "__main__":
    detect()
```

```html
<script>
const POSITION_JUMP = 5;    // gleiche Schwellen wie detect_changes.py
const TOTAL_PCT     = 0.15;
const TOP_SHOW      = 10;

async function loadData() {
  const idx = await (await fetchJson('data/index.json')).json();
  const dates = idx.dates || [];
  if (!dates.length) throw new Error('Noch keine Snapshots im Repo.');
  const today = await (await fetchJson(`data/${dates[0]}.json`)).json();
  let yesterday = null;
  if (dates.length >= 2) {
    yesterday = await (await fetchJson(`data/${dates[1]}.json`)).json();
  }
  return { today, yesterday };
}

async function fetchJson(path) {
  const r = await fetch(path, { cache: 'no-cache' });
  if (!r.ok) throw new Error(`${path}: HTTP ${r.status}`);
  return r;
}

function computeAlerts(today, yesterday) {
  const jumps = [], newTop5 = [], totalShifts = [];
  if (!yesterday) return { jumps, newTop5, totalShifts };
  for (const [q, d] of Object.entries(today.queries)) {
    const old = yesterday.queries[q];
    if (!old) continue;
    if (old.total_results && d.total_results) {
      const change = (d.total_results - old.total_results) / old.total_results;
      if (Math.abs(change) >= TOTAL_PCT) {
        totalShifts.push({ q, old: old.total_results, neu: d.total_results, pct: change });
      }
    }
    const ydayMap = new Map(old.results.map(r => [r.angebot_id, r.position]));
    for (const r of d.results) {
      const op = ydayMap.get(r.angebot_id);
      if (op === undefined) {
        if (r.position <= 5) newTop5.push({ q, titel: r.titel, pos: r.position });
      } else {
        const diff = op - r.position;
        if (Math.abs(diff) >= POSITION_JUMP) {
          jumps.push({ q, titel: r.titel, oldPos: op, newPos: r.position });
        }
      }
    }
  }
  return { jumps, newTop5, totalShifts };
}

function renderAlerts(a, hasYesterday) {
  const box  = document.getElementById('alerts');
  const body = document.getElementById('alerts-body');
  box.hidden = false;
  if (!hasYesterday) {
    body.innerHTML = '<p>Noch keine Vergleichsbasis — es gibt erst einen Snapshot. Auffälligkeiten erscheinen ab dem zweiten Lauf.</p>';
    return;
  }
  const total = a.jumps.length + a.newTop5.length + a.totalShifts.length;
  if (total === 0) {
    body.innerHTML = '<p>Keine Auffälligkeiten über den Schwellen. Markt stabil.</p>';
    return;
  }
  const head = `<p><strong>${a.jumps.length}</strong> Positionssprünge · <strong>${a.newTop5.length}</strong> neue Top-5 · <strong>${a.totalShifts.length}</strong> Trefferzahl-Wechsel</p>`;
  const lines = [];
  for (const x of a.jumps) {
    const arrow = x.newPos < x.oldPos ? '▲' : '▼';
    const cls   = x.newPos < x.oldPos ? 'up' : 'down';
    lines.push(`<div class="alert"><span class="${cls}">[SPRUNG ${arrow}]</span> '${escapeHtml(x.q)}': ${escapeHtml(x.titel)} — Platz ${x.oldPos} → ${x.newPos}</div>`);
  }
  for (const x of a.newTop5) {
    lines.push(`<div class="alert"><span class="new">[NEU TOP5]</span> '${escapeHtml(x.q)}': ${escapeHtml(x.titel)} — Platz ${x.pos}</div>`);
  }
  for (const x of a.totalShifts) {
    const sign = x.pct > 0 ? '+' : '';
    lines.push(`<div class="alert">[TREFFERZAHL] '${escapeHtml(x.q)}': ${x.old} → ${x.neu} (${sign}${(x.pct*100).toFixed(0)} %)</div>`);
  }
  body.innerHTML = head + lines.join('');
}

function renderLists(today, yesterday) {
  const root = document.getElementById('lists');
  const queries = Object.keys(today.queries).sort();
  const html = [];
  for (const q of queries) {
    const d = today.queries[q];
    const old = yesterday && yesterday.queries ? yesterday.queries[q] : null;
    const ydayMap = new Map((old?.results || []).map(r => [r.angebot_id, r.position]));
    const totalLabel = d.total_results != null
      ? `${d.total_results.toLocaleString('de-DE')} Treffer` : '';
    html.push(`<details open><summary>${escapeHtml(q)} <span class="total">— ${totalLabel}</span></summary>`);
    html.push('<table><tbody>');
    for (const r of d.results.slice(0, TOP_SHOW)) {
      const op = ydayMap.get(r.angebot_id);
      let delta = '<span class="anb">·</span>';
      if (old && op === undefined)              delta = `<span class="new">NEU</span>`;
      else if (op !== undefined && op > r.position) delta = `<span class="up">▲ ${op - r.position}</span>`;
      else if (op !== undefined && op < r.position) delta = `<span class="down">▼ ${r.position - op}</span>`;
      html.push(`<tr>
        <td class="pos">${r.position}.</td>
        <td>${escapeHtml(r.titel || '')}<br><span class="anb">${escapeHtml(r.anbieter || '')}</span></td>
        <td class="delta">${delta}</td>
      </tr>`);
    }
    html.push('</tbody></table></details>');
  }
  root.innerHTML = html.join('');
}

function escapeHtml(s) {
  return String(s).replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));
}

(async () => {
  try {
    const { today, yesterday } = await loadData();
    const alerts = computeAlerts(today, yesterday);
    renderAlerts(alerts, !!yesterday);
    renderLists(today, yesterday);
    document.getElementById('meta').textContent =
```

## Task 10: GitHub Actions Workflow + manueller Trigger

- [ ] **Step 10.1: Workflow-Datei anlegen**

Inhalt von `.github/workflows/daily.yml`:

```yaml
name: Daily Snapshot

on:
  schedule:
    - cron: '0 6 * * *'   # täglich 06:00 UTC
  workflow_dispatch:       # erlaubt manuellen Trigger über die GH UI

permissions:
  contents: write          # damit der Job das Snapshot-JSON zurück-committen kann

concurrency:
  group: daily-snapshot
  cancel-in-progress: false

jobs:
  snapshot:
    runs-on: ubuntu-latest
    timeout-minutes: 5
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Run snapshot
        run: python run_local.py

      - name: Commit snapshot
        run: |
          git config user.name  "github-actions[bot]"
          git config user.email "41898282+github-actions[bot]@users.noreply.github.com"
          git add docs/data/
          if git diff --cached --quiet; then
            echo "No changes to commit."
          else
            git commit -m "data: snapshot $(date -u +%Y-%m-%d)"
            git push
          fi
```
